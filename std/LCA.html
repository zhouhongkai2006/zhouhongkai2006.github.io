题目描述

如题，给定一棵有根多叉树，请求出指定两个点直接最近的公共祖先。

输入输出格式

输入格式：

第一行包含三个正整数N、M、S，分别表示树的结点个数、询问的个数和树根结点的序号。

接下来N-1行每行包含两个正整数x、y，表示x结点和y结点之间有一条直接连接的边（数据保证可以构成树）。

接下来M行每行包含两个正整数a、b，表示询问a结点和b结点的最近公共祖先。

输出格式：

输出包含M行，每行包含一个正整数，依次为每一个询问的结果。

输入输出样例

输入样例#1：

5 5 4

3 1

2 4

5 1

1 4

2 4

3 2

3 5

1 2

4 5

输出样例#1：

4

4

1

4

4

说明时空限制：1000ms,128M

数据规模：对于30%的数据：N<=10，M<=10

          对于70%的数据：N<=10000，M<=10000

          对于100%的数据：N<=500000，M<=500000

样例说明：

该树结构如下：

第一次询问：2、4的最近公共祖先，故为4。

第二次询问：3、2的最近公共祖先，故为4。

第三次询问：3、5的最近公共祖先，故为1。

第四次询问：1、2的最近公共祖先，故为4。

第五次询问：4、5的最近公共祖先，故为4。

故输出依次为4、4、1、4、4。

题解：标准LCA的倍增算法或者使用Tarjan算法。

  2. #include <cstdio>
  3. #include <cmath>
  4. #include <algorithm>
  5. #define N 1000005
  6. int n,m,s,x,y,cnt;
  7. int first[N],next[N],v[N],dep[N/2],f[N][25];
  8. bool vis[N/2];
  9. using namespace std;
  10. inline int read()
  11. {
  12. 	int f=1,x=0;
  13. 	char ch=getchar();
  14. 	if (ch=='-')
  15. 	{
  16. 		f=-1;
  17. 		ch=getchar();
  18. 	}
  19. 	while ((ch<'0')||(ch>'9')) ch=getchar();
  20. 	while ((ch>='0')&&(ch<='9'))
  21. 	{
  22. 		x=x*10+ch-'0';
  23. 		ch=getchar();
  24. 	}
  25. 	return f*x;
  26. }
  27. inline void dfs(int root,int k)
  28. {
  29. 	dep[root]=k;
  30. 	vis[root]=1;
  31. 	for (int i=first[root];i;i=next[i]) 
  32. 	  if (!vis[v[i]])
  33. 	  {
  34. 		  f[v[i]][0]=root;	
  35. 		  dfs(v[i],k+1);
  36. 	  }
  37. }
  38. inline int find(int x,int y)
  39. {
  40. 	if (dep[x]<dep[y]) swap(x,y);
  41. 	for (int i=log2(n);i>=0;i--)
  42. 	  if (dep[f[x][i]]>=dep[y]) x=f[x][i];
  43. 	if (x==y) return x;
  44. 	for (int i=log2(n);i>=0;i--)
  45. 	  if (f[x][i]!=f[y][i])
  46. 	  {
  47. 	  	x=f[x][i];
  48. 	  	y=f[y][i];
  49. 	  }
  50. 	  return f[x][0];
  51. }
  52. int main()
  53. {
  54. 	freopen("1.in","r",stdin);
  55. 	n=read(),m=read(),s=read();
  56. 	for (int i=1;i<n;i++)
  57. 	{
  58. 		x=read(),y=read();
  59. 		next[++cnt]=first[x];
  60. 		first[x]=cnt;
  61. 		v[cnt]=y;
  62. 		next[++cnt]=first[y];
  63. 		first[y]=cnt;
  64. 		v[cnt]=x;
  65. 	}
  66. 	dfs(s,1);
  67. 	for (int j=1;j<=log2(n);j++)
  68. 	  for (int i=1;i<=n;i++)
  69. 	    f[i][j]=f[f[i][j-1]][j-1];
  70. 	for (int i=1;i<=m;i++)
  71. 	{
  72. 		x=read(),y=read();
  73. 		if ((x==s)||(y==s)) printf("%d\n",s); else
  74. 		  printf("%d\n",find(x,y));
  75. 	}
  76. 	return 0;
  77. }

  2. #include <cstdio>
  3. #define N 1000005
  4. int n,m,s,x,y,cnt;
  5. int first[N],next[N],v[N],f[N/2],answer[N/2];
  6. int first1[N],next1[N],v1[N],id[N];
  7. bool vis[N/2];
  8. using namespace std;
  9. inline int read()
  10. {
  11.     int f=1,x=0;
  12.     char ch=getchar();
  13.     if (ch=='-')
  14.     {
  15.         f=-1;
  16.         ch=getchar();
  17.     }
  18.     while ((ch<'0')||(ch>'9')) ch=getchar();
  19.     while ((ch>='0')&&(ch<='9'))
  20.     {
  21.         x=x*10+ch-'0';
  22.         ch=getchar();
  23.     }
  24.     return f*x;
  25. }
  26. inline int find(int x)
  27. {
  28.     if (f[x]==x) return x;
  29.     int t=find(f[x]);
  30.     f[x]=t;
  31.     return t;
  32. }
  33. inline void hb(int x,int y)
  34. {
  35.     int xx=find(x),yy=find(y);
  36.     if (xx!=yy) f[xx]=yy;
  37. }
  38. inline void tarjan(int root)
  39. {
  40.     vis[root]=1;
  41.     for (int i=first1[root];i;i=next1[i])
  42.       if (vis[v1[i]]) answer[id[i]]=find(v1[i]);
  43.     for (int i=first[root];i;i=next[i])
  44.       if (!vis[v[i]]) 
  45.       {
  46.       	tarjan(v[i]);
  47.       	hb(v[i],root);
  48.       }
  49. }
  50. int main()
  51. {
  52.     n=read(),m=read(),s=read();
  53.     for (int i=1;i<n;i++)
  54.     {
  55.         x=read(),y=read();
  56.         next[++cnt]=first[x];
  57.         first[x]=cnt;
  58.         v[cnt]=y;
  59.         next[++cnt]=first[y];
  60.         first[y]=cnt;
  61.         v[cnt]=x;
  62.     }
  63.     cnt=0;
  64.     for (int i=1;i<=m;i++)
  65.     {
  66.         x=read(),y=read();
  67.         next1[++cnt]=first1[x];
  68.         first1[x]=cnt;
  69.         v1[cnt]=y;
  70.         id[cnt]=i;
  71.         next1[++cnt]=first1[y];
  72.         first1[y]=cnt;
  73.         v1[cnt]=x;
  74.         id[cnt]=i;
  75.     }
  76.     for (int i=1;i<=n;i++) f[i]=i;
  77.     tarjan(s);
  78.     for (int i=1;i<=m;i++) printf("%d\n",answer[i]);
  79.     return 0;
  80. }
